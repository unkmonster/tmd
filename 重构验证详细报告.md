```go
func executeUserDownload(ctx context.Context, deps *Dependencies, screenName string, opts DownloadOptions, pathHelper *StorePath) error {
    // 1. 获取用户信息
    user, uid, err := twitter.GetUserByScreenName(ctx, deps.Client, screenName)
    if err != nil {
        database.MarkUserInaccessible(deps.DB, uid, screenName)
        return err
    }
    
    // 2. 初始化下载器
    versionManager := downloader.NewVersionManagerWithWriter(".versions", nil)
    fileWriter := downloader.NewFileWriter(versionManager)
    dwn := downloader.NewDownloader(fileWriter)
    
    // 3. 初始化 Dumper
    dumper := downloading.NewDumper()
    _ = dumper.Load(pathHelper.ErrorJ)
    defer func() {
        if dumper.Count() > 0 {
            dumper.Dump(pathHelper.ErrorJ)
        }
    }()
    
    // 4. 执行批量下载
    failedTweets, _, err := downloading.BatchDownloadAny(ctx, deps.Client, deps.DB, nil, []*twitter.User{user}, pathHelper.Root, pathHelper.Users, opts.AutoFollow, deps.AdditionalClients, dwn, fileWriter)
    
    // 5. 收集失败的推文
    for _, tweet := range failedTweets {
        if tweet.Entity != nil {
            if id, err := tweet.Entity.Id(); err == nil {
                dumper.Push(id, tweet.Tweet)
            }
        }
    }
    
    // 6. 重试
    if !opts.NoRetry {
        if err := downloading.RetryFailedTweets(ctx, dumper, deps.DB, deps.Client, dwn, fileWriter); err != nil {
            log.Warnf("Retry failed tweets error: %v", err)
        }
    }
    
    // 7. Profile 下载
    if !opts.SkipProfile {
        // ... Profile 下载逻辑
    }
    
    return nil
}
```

```go
func (s *downloadServiceImpl) UserDownload(ctx context.Context, taskID string, screenName string, opts DownloadOptions, reporter ProgressReporter) error {
    if reporter == nil {
        reporter = &NopReporter{}
    }
    
    reporter.OnProgress(taskID, Progress{Stage: "downloading", Current: screenName})
    
    // 1. 获取存储路径
    pathHelper, err := path.NewStorePath(s.deps.Config.RootPath)
    if err != nil {
        return fmt.Errorf("failed to make store dir: %w", err)
    }
    
    // 2. 初始化 Dumper
    dumper := downloading.NewDumper()
    if err := dumper.Load(pathHelper.ErrorJ); err != nil {
        log.Warnf("Failed to load dumper: %v", err)
    }
    defer s.saveDumper(dumper, pathHelper.ErrorJ)
    
    // 3. 获取用户信息
    user, uid, err := twitter.GetUserByScreenName(ctx, s.deps.Client, screenName)
    if err != nil {
        s.userRepo.MarkInaccessible(ctx, uid, screenName)  // 使用 Repository 层
        return err
    }
    
    // 4. 初始化下载器
    versionManager := downloader.NewVersionManagerWithWriter(".versions", nil)
    fileWriter := downloader.NewFileWriter(versionManager)
    dwn := downloader.NewDownloader(fileWriter)
    
    // 5. 执行批量下载
    failedTweets, _, err := downloading.BatchDownloadAny(ctx, s.deps.Client, s.deps.DB, nil, []*twitter.User{user}, pathHelper.Root, pathHelper.Users, opts.AutoFollow, s.deps.AdditionalClients, dwn, fileWriter)
    if err != nil {
        return err
    }
    
    // 6. 收集失败的推文
    s.collectFailedTweets(dumper, failedTweets)
    
    // 7. 重试
    if !opts.NoRetry {
        if err := downloading.RetryFailedTweets(ctx, dumper, s.deps.DB, s.deps.Client, dwn, fileWriter); err != nil {
            log.Warnf("Retry failed tweets error: %v", err)
        }
    }
    
    // 8. Profile 下载
    if !opts.SkipProfile {
        if err := s.downloadProfile(ctx, taskID, []*twitter.User{user}, pathHelper, versionManager, fileWriter, dwn, reporter, opts.SkipProfile); err != nil {
            log.Warnf("Profile download failed for %s: %v", screenName, err)
        }
    }
    
    reporter.OnComplete(taskID, Result{Message: "User download completed"})
    return nil
}
```

### 4.1 错误处理

**稳定版**:
```go
func MarkUserInaccessible(db *sqlx.DB, uid uint64, screenName string) {
    if uid == 0 {
        return  // 直接返回，不返回错误
    }
    
    _, err := db.Exec(...)
    if err != nil {

}
```

**重构版**:
```go
func (r *userRepository) MarkInaccessible(ctx context.Context, uid uint64, screenName string) error {
    if uid == 0 {
        return nil  // 返回 nil 表示成功（空操作）
    }
    
    _, err := r.db.ExecContext(ctx, ...)
    if err != nil {
        log.Warnf("Failed to mark user %s as inaccessible: %v", screenName, err)
        return err  // 返回错误，由调用者决定如何处理
    }
    return nil
}
```

**差异影响**:
- ⚠️ 行为变更：调用者现在可以感知错误
- ✅ 改进：符合 Go 错误处理最佳实践
